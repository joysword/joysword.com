159  FINGER_3_ROTATE_MIDDLE,
160  FINGER_4_ROTATE_MIDDLE,
161 
162  // rotation at outer knuckle
163  FINGER_0_ROTATE_OUTER,
164  FINGER_1_ROTATE_OUTER,
165  FINGER_2_ROTATE_OUTER,
166  FINGER_3_ROTATE_OUTER,
167  FINGER_4_ROTATE_OUTER,

169  LAST_ACTUAL_ARTICULATION,
170 
171  //
172  // Vistual articulations
173  //
174 
175  // For fingers, data glove gets one angle for both the
176  // middle and outer knucles. These virtual articulations

178  // outer knuckles.

180  FINGER_2_MIDDLE_OUTER,
181  FINGER_3_MIDDLE_OUTER,
182  FINGER_4_MIDDLE_OUTER,
183  // NOTE: FINGER_0 (thumb) isn't included because the
184  // glove provides angles for both the middle and outer
185  // knuckles.
186 
187  // Spread angles. Setting these virtual articulations will
188  // override the "lateral rotation / flexure" articulations.
189  SPREAD_0_1,
190  SPREAD_1_2,
191  SPREAD_2_3,
192  SPREAD_3_4,
193 
194  MAX_ARTICULATIONS
